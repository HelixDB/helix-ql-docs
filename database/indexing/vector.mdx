---
title: 'Vector Indexes'
---

> For the complete documentation index optimized for AI agents, see [llms.txt](/llms.txt).

Vector indexes enable approximate nearest-neighbor search on numeric array properties. They are
available on nodes and edges, normalize values to float32, and support cosine, euclidean, and
manhattan distance. Query vectors must match the indexed vectors' dimension.

Vector properties must be top-level. The optional tenant property partitions an index; pass no
tenant property for a global index, or see [Multi-Tenancy](/database/multi-tenancy).

## Create Vector Indexes

Index creation is an idempotent write route:

<CodeGroup>
```ts TypeScript
import { g, IndexSpec, writeBatch } from "@helix-db/helix-db";

const createVectorIndexes = writeBatch()
  .varAs("doc_embedding", g().createIndexIfNotExists(IndexSpec.nodeVector("Doc", "embedding")))
  .varAs("citation_embedding", g().createIndexIfNotExists(IndexSpec.edgeVector("CITES", "embedding")));
```

```rust Rust
use helix_db::dsl::prelude::*;

#[register]
pub fn create_vector_indexes() -> WriteBatch {
    write_batch()
        .var_as(
            "doc_embedding",
            g().create_index_if_not_exists(IndexSpec::node_vector("Doc", "embedding", None::<&str>)),
        )
        .var_as(
            "citation_embedding",
            g().create_index_if_not_exists(IndexSpec::edge_vector("CITES", "embedding", None::<&str>)),
        )
}
```

```go Go
import helix "github.com/helixdb/helix-db/sdks/go"

func CreateVectorIndexes() helix.Request {
	return helix.WriteQuery("create_vector_indexes").
		VarAs("doc_embedding", helix.G().CreateIndexIfNotExists(helix.NodeVectorIndex("Doc", "embedding"))).
		VarAs("citation_embedding", helix.G().CreateIndexIfNotExists(helix.EdgeVectorIndex("CITES", "embedding"))).
		Returning()
}
```


## Search Nodes and Edges

The search source returns the nearest `k` hits in ascending distance order:

<CodeGroup>
```ts TypeScript
import { g, readBatch } from "@helix-db/helix-db";

const vectorSearch = readBatch()
  .varAs("docs", g().vectorSearchNodes("Doc", "embedding", [0.12, 0.85, -0.04], 10, null))
  .varAs("citations", g().vectorSearchEdges("CITES", "embedding", [0.12, 0.85, -0.04], 10, null))
  .returning(["docs", "citations"]);
```

```rust Rust
use helix_db::dsl::prelude::*;

#[register]
pub fn vector_search() -> ReadBatch {
    read_batch()
        .var_as(
            "docs",
            g().vector_search_nodes("Doc", "embedding", vec![0.12f32, 0.85, -0.04], 10, None::<PropertyValue>),
        )
        .var_as(
            "citations",
            g().vector_search_edges("CITES", "embedding", vec![0.12f32, 0.85, -0.04], 10, None::<PropertyValue>),
        )
        .returning(["docs", "citations"])
}
```

```go Go
import helix "github.com/helixdb/helix-db/sdks/go"

func VectorSearch() helix.Request {
	return helix.ReadQuery("vector_search").
		VarAs("docs", helix.G().VectorSearchNodes("Doc", "embedding", []float32{0.12, 0.85, -0.04}, 10)).
		VarAs("citations", helix.G().VectorSearchEdges("CITES", "embedding", []float32{0.12, 0.85, -0.04}, 10)).
		Returning("docs", "citations")
}
```


stream if the response needs the score.

## Parameterized Search

Use the `_with` form when the vector and result limit come from request parameters:

<CodeGroup>
```ts TypeScript
import { defineParams, Expr, g, param, PropertyInput, readBatch, StreamBound } from "@helix-db/helix-db";

const params = defineParams({ query_vector: param.array(param.f32()), limit: param.i64() });

function semanticSearch(p = params) {
  return readBatch()
    .varAs(
      "docs",
      g().vectorSearchNodesWith(
        "Doc",
        "embedding",
        PropertyInput.param("query_vector"),
        StreamBound.expr(Expr.param("limit")),
        null,
      ),
    )
    .returning(["docs"]);
}
```

```rust Rust
use helix_db::dsl::prelude::*;

#[register]
pub fn semantic_search(query_vector: Vec<f32>, limit: i64) -> ReadBatch {
    let _ = (&query_vector, &limit);
    read_batch()
        .var_as(
            "docs",
            g().vector_search_nodes_with(
                "Doc",
                "embedding",
                PropertyInput::param("query_vector"),
                Expr::param("limit"),
                None::<PropertyInput>,
            ),
        )
        .returning(["docs"])
}
```

```go Go
import helix "github.com/helixdb/helix-db/sdks/go"

func SemanticSearch(queryVector []float32, limit int64) helix.Request {
	q := helix.ReadQuery("semantic_search")
	vector := q.ParamArray("query_vector", queryVector, helix.ParamTypeF32())
	k := q.ParamI64("limit", limit)

	return q.
		VarAs("docs", helix.G().VectorSearchNodesWith("Doc", "embedding", vector.Input(), k.Bound(), nil)).
		Returning("docs")
}
```

```python Python
from helixdb import define_params, g, param, read_batch

params = define_params({"query_vector": param.array(param.f32()), "limit": param.i64()})

def semantic_search(p=params):
    return (
        read_batch()
        .var_as("docs", g().vector_search_nodes_with("Doc", "embedding", p.query_vector.input(), p.limit, None))
        .returning(["docs"])
    )
```
