> For the complete documentation index optimized for AI agents, see [llms.txt](/llms.txt).

Secondary indexes accelerate property-based filtering on the labeled property graph. They are
available on both nodes and edges:

- **Equality indexes** map `property = value` to matching IDs.
- **Range indexes** provide ordered scans for numeric and string comparisons.

Indexes are opt-in. They speed up reads at the cost of write overhead and storage. Only top-level
properties can be indexed; dotted paths remain scan-only in V1.

## Create Secondary Indexes

Index creation is an idempotent write route. This example creates node and edge variants for both
index families:

<CodeGroup>
```ts TypeScript
import { g, IndexSpec, writeBatch } from "@helix-db/helix-db";

const createSecondaryIndexes = writeBatch()
  .varAs("user_status", g().createIndexIfNotExists(IndexSpec.nodeEquality("User", "status")))
  .varAs("follows_since", g().createIndexIfNotExists(IndexSpec.edgeEquality("FOLLOWS", "since_year")))
  .varAs("event_time", g().createIndexIfNotExists(IndexSpec.nodeRange("Event", "timestamp")))
  .varAs("rating_score", g().createIndexIfNotExists(IndexSpec.edgeRange("RATED", "score")));
```

```rust Rust
use helix_db::dsl::prelude::*;

#[register]
pub fn create_secondary_indexes() -> WriteBatch {
    write_batch()
        .var_as(
            "user_status",
            g().create_index_if_not_exists(IndexSpec::node_equality("User", "status")),
        )
        .var_as(
            "follows_since",
            g().create_index_if_not_exists(IndexSpec::edge_equality("FOLLOWS", "since_year")),
        )
        .var_as(
            "event_time",
            g().create_index_if_not_exists(IndexSpec::node_range("Event", "timestamp")),
        )
        .var_as(
            "rating_score",
            g().create_index_if_not_exists(IndexSpec::edge_range("RATED", "score")),
        )
}
```

```go Go
import helix "github.com/helixdb/helix-db/sdks/go"

func CreateSecondaryIndexes() helix.Request {
	return helix.WriteQuery("create_secondary_indexes").
		VarAs("user_status", helix.G().CreateIndexIfNotExists(helix.NodeEqualityIndex("User", "status"))).
		VarAs("follows_since", helix.G().CreateIndexIfNotExists(helix.EdgeEqualityIndex("FOLLOWS", "since_year"))).
		VarAs("event_time", helix.G().CreateIndexIfNotExists(helix.NodeRangeIndex("Event", "timestamp"))).
		VarAs("rating_score", helix.G().CreateIndexIfNotExists(helix.EdgeRangeIndex("RATED", "score"))).
		Returning()
}
```

Range indexes use ascending physical order by default. For newest-first or high-score-first access,
use `nodeRangeDesc` / `node_range_desc` / `NodeRangeDescIndex` and the matching edge helpers.

## Equality Lookups

Use the source-predicate form to push equality filters into the index:

<CodeGroup>
```ts TypeScript
import { g, readBatch, SourcePredicate } from "@helix-db/helix-db";

const equalityLookups = readBatch()
  .varAs("active_users", g().nWithLabelWhere("User", SourcePredicate.eq("status", "active")))
  .varAs("recent_follows", g().eWithLabelWhere("FOLLOWS", SourcePredicate.eq("since_year", 2024)))
  .returning(["active_users", "recent_follows"]);
```

```rust Rust
use helix_db::dsl::prelude::*;

#[register]
pub fn equality_lookups() -> ReadBatch {
    read_batch()
        .var_as(
            "active_users",
            g().n_with_label_where("User", SourcePredicate::eq("status", "active")),
        )
        .var_as(
            "recent_follows",
            g().e_with_label_where("FOLLOWS", SourcePredicate::eq("since_year", 2024i64)),
        )
        .returning(["active_users", "recent_follows"])
}
```

```go Go
import helix "github.com/helixdb/helix-db/sdks/go"

func EqualityLookups() helix.Request {
	return helix.ReadQuery("equality_lookups").
		VarAs("active_users", helix.G().NWithLabelWhere("User", helix.SourceEq("status", "active"))).
		VarAs("recent_follows", helix.G().EWithLabelWhere("FOLLOWS", helix.SourceEq("since_year", int64(2024)))).
		Returning("active_users", "recent_follows")
}
```

```python Python
from helixdb import SourcePredicate, g, read_batch

equality_lookups = (
    read_batch()
    .var_as("active_users", g().n_with_label_where("User", SourcePredicate.eq("status", "active")))
    .var_as("recent_follows", g().e_with_label_where("FOLLOWS", SourcePredicate.eq("since_year", 2024)))
    .returning(["active_users", "recent_follows"])
)
```

</CodeGroup>

## Range Lookups

Range indexes support `gt`, `gte`, `lt`, `lte`, and `between` source predicates:

<CodeGroup>
```ts TypeScript
import { g, readBatch, SourcePredicate } from "@helix-db/helix-db";

const rangeLookups = readBatch()
  .varAs("recent_events", g().nWithLabelWhere("Event", SourcePredicate.gt("timestamp", 1_700_000_000)))
  .varAs("high_ratings", g().eWithLabelWhere("RATED", SourcePredicate.gte("score", 4)))
  .returning(["recent_events", "high_ratings"]);
```

```rust Rust
use helix_db::dsl::prelude::*;

#[register]
pub fn range_lookups() -> ReadBatch {

        )

            g().e_with_label_where("RATED", SourcePredicate::gte("score", 4i64)),

}
```
